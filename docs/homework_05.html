random_value <- runif(1, min=3, max=10)
n_dims <- as.integer(random_value)
print(n_dims)
## [1] 3
# value = 3

my_vec <- 1:n_dims^2

sample(x=my_vec)
## [1] 6 2 7 9 3 8 4 5 1
m <- matrix(data=my_vec,nrow=n_dims)
print(m)
##      [,1] [,2] [,3]
## [1,]    1    4    7
## [2,]    2    5    8
## [3,]    3    6    9
transposed_matrix <- t(m)
print(transposed_matrix)
##      [,1] [,2] [,3]
## [1,]    1    2    3
## [2,]    4    5    6
## [3,]    7    8    9
# switches order of columns and rows

rowSums(m)
## [1] 12 15 18
# values = 12,15,18

rowMeans(m)
## [1] 4 5 6
# values = 4,5,6

eigen(m)
